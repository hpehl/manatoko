44  import static org.jboss.hal.testsuite.fixtures.MessagingFixtures.CALL_TIMEOUT;
45  import static org.jboss.hal.testsuite.fixtures.MessagingFixtures.CC_CREATE;
46  import static org.jboss.hal.testsuite.fixtures.MessagingFixtures.CC_DELETE;
47  import static org.jboss.hal.testsuite.fixtures.MessagingFixtures.CC_UPDATE;
48  import static org.jboss.hal.testsuite.fixtures.MessagingFixtures.CC_UPDATE_ALTERNATIVES;
49  import static org.jboss.hal.testsuite.fixtures.MessagingFixtures.CLUSTER_CONNECTION_ADDRESS;
50  import static org.jboss.hal.testsuite.fixtures.MessagingFixtures.SRV_UPDATE;
51  import static org.jboss.hal.testsuite.fixtures.MessagingFixtures.clusterConnectionAddress;
52  
53  @Manatoko
54  @Testcontainers
55  class ClusterConnectionTest extends AbstractClusteringTest {
56  
57      private static final Values CC_PARAMS = Values.of(CLUSTER_CONNECTION_ADDRESS, Random.name())
58              .and(CONNECTOR_NAME, HTTP_CONNECTOR)
59              .and(DISCOVERY_GROUP, Random.name());
60  
61      @Container static WildFlyContainer wildFly = WildFlyContainer.standalone(FULL_HA);
62  
63      @BeforeAll
64      static void setupModel() throws Exception {
65          OnlineManagementClient client = wildFly.managementClient();
66          client.apply(new AddMessagingServer(SRV_UPDATE));
67          Operations operations = new Operations(client);
68          operations.add(clusterConnectionAddress(SRV_UPDATE, CC_UPDATE), CC_PARAMS).assertSuccess();
69          operations.add(clusterConnectionAddress(SRV_UPDATE, CC_UPDATE_ALTERNATIVES), CC_PARAMS).assertSuccess();
70          operations.add(clusterConnectionAddress(SRV_UPDATE, CC_DELETE), CC_PARAMS).assertSuccess();
71      }
72  
73      @BeforeEach
74      void setUp() {
75          page.navigate(SERVER, SRV_UPDATE);
76      }
77  
78      @Test
79      void clusterConnectionCreate() throws Exception {
80          console.verticalNavigation().selectPrimary(Ids.build(MESSAGING_CLUSTER_CONNECTION, ITEM));
81          console.waitNoNotification();
82          TableFragment table = page.getClusterConnectionTable();
83          FormFragment form = page.getClusterConnectionForm();
84          table.bind(form);
85          crudOperations.create(clusterConnectionAddress(SRV_UPDATE, CC_CREATE), table, f -> {
86              f.text(NAME, CC_CREATE);
87              f.text(CLUSTER_CONNECTION_ADDRESS, Random.name());
88              f.text(CONNECTOR_NAME, HTTP_CONNECTOR);
89              f.text(DISCOVERY_GROUP, Random.name());
90          });
91      }
92  
93      @Test
94      void clusterConnectionTryCreate() {
95          page.navigateAgain(SERVER, SRV_UPDATE);
96          console.verticalNavigation().selectPrimary(Ids.build(MESSAGING_CLUSTER_CONNECTION, ITEM));
97          TableFragment table = page.getClusterConnectionTable();
98          FormFragment form = page.getClusterConnectionForm();
99          table.bind(form);
100         crudOperations.createWithErrorAndCancelDialog(table, f -> f.text(NAME, CC_CREATE), CLUSTER_CONNECTION_ADDRESS);
101     }
102 
103     @Test
104     void clusterConnectionUpdate() throws Exception {
105         console.verticalNavigation().selectPrimary(Ids.build(MESSAGING_CLUSTER_CONNECTION, ITEM));
106         TableFragment table = page.getClusterConnectionTable();
107         FormFragment form = page.getClusterConnectionForm();
108         table.bind(form);
109         table.scrollToTop();
110         table.select(CC_UPDATE);
111         crudOperations.update(clusterConnectionAddress(SRV_UPDATE, CC_UPDATE), form, CALL_TIMEOUT, 123L);
112     }
113 
114     @Test
115     void clusterConnectionTryUpdateAlternatives() {
116         console.verticalNavigation().selectPrimary(Ids.build(MESSAGING_CLUSTER_CONNECTION, ITEM));
117         TableFragment table = page.getClusterConnectionTable();
118         FormFragment form = page.getClusterConnectionForm();
119         table.bind(form);
120         table.scrollToTop();
121         table.select(CC_UPDATE_ALTERNATIVES);
122 
123         crudOperations.updateWithError(form, f -> {
124             f.text(DISCOVERY_GROUP, Random.name());
125             f.list(STATIC_CONNECTORS).add(HTTP);
126         }, STATIC_CONNECTORS);
127     }
128 
129     @Test
130     void clusterConnectionRemove() throws Exception {
131         console.verticalNavigation().selectPrimary(Ids.build(MESSAGING_CLUSTER_CONNECTION, ITEM));
132         TableFragment table = page.getClusterConnectionTable();
133         FormFragment form = page.getClusterConnectionForm();
134         table.bind(form);
135 
136         crudOperations.delete(clusterConnectionAddress(SRV_UPDATE, CC_DELETE), table, CC_DELETE);
137     }
138 }
