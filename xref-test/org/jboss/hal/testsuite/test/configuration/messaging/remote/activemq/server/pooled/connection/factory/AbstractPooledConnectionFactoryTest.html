16  package org.jboss.hal.testsuite.test.configuration.messaging.remote.activemq.server.pooled.connection.factory;
17  
18  import java.io.IOException;
19  
20  import org.jboss.arquillian.core.api.annotation.Inject;
21  import org.jboss.arquillian.graphene.page.Page;
22  import org.jboss.hal.dmr.ModelDescriptionConstants;
23  import org.jboss.hal.testsuite.Console;
24  import org.jboss.hal.testsuite.CrudOperations;
25  import org.jboss.hal.testsuite.Random;
26  import org.jboss.hal.testsuite.fixtures.MessagingFixtures;
27  import org.jboss.hal.testsuite.model.ModelNodeGenerator;
28  import org.jboss.hal.testsuite.page.configuration.MessagingRemoteActiveMQPage;
29  import org.wildfly.extras.creaper.core.online.operations.Batch;
30  import org.wildfly.extras.creaper.core.online.operations.Operations;
31  import org.wildfly.extras.creaper.core.online.operations.Values;
32  
33  abstract class AbstractPooledConnectionFactoryTest {
34  
35      protected static void createDiscoveryGroup(Operations operations, String name, String jgroupsChannel)
36              throws IOException {
37          Batch batch = new Batch();
38          batch.add(MessagingFixtures.RemoteActiveMQServer.discoveryGroupAddress(name));
39          batch.writeAttribute(MessagingFixtures.RemoteActiveMQServer.discoveryGroupAddress(name), "jgroups-channel",
40                  jgroupsChannel);
41          batch.writeAttribute(
42                  MessagingFixtures.RemoteActiveMQServer.discoveryGroupAddress(name), "jgroups-cluster", Random.name());
43          operations.batch(batch).assertSuccess();
44      }
45  
46      protected static void createPooledConnectionFactory(Operations operations, String name, String discoveryGroup)
47              throws IOException {
48          operations.add(MessagingFixtures.RemoteActiveMQServer.pooledConnectionFactoryAddress(name),
49                  Values.of(ModelDescriptionConstants.DISCOVERY_GROUP, discoveryGroup)
50                          .and("entries",
51                                  new ModelNodeGenerator.ModelNodeListBuilder()
52                                          .addAll(Random.name(), Random.name(), Random.name())
53                                          .build()))
54                  .assertSuccess();
55      }
56  
57      protected static void createGenericConnector(Operations operations, String name) throws IOException {
58          operations.add(
59                  MessagingFixtures.RemoteActiveMQServer.genericConnectorAddress(name),
60                  Values.of("factory-class", Random.name()))
61                  .assertSuccess();
62      }
63  
64      @Inject protected Console console;
65      @Inject protected CrudOperations crudOperations;
66      @Page protected MessagingRemoteActiveMQPage page;
67  }
