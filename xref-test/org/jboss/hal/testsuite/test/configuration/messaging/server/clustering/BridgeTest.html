219     @Test
220     void bridgeAddCredentialReference() throws Exception {
221         operations.undefineAttribute(bridgeAddress(SRV_UPDATE, BRIDGE_UPDATE), PASSWORD);
222         operations.undefineAttribute(bridgeAddress(SRV_UPDATE, BRIDGE_UPDATE), CREDENTIAL_REFERENCE);
223         // navigate again, to reload the page as new data were added with the operations above
224 
225         page.navigateAgain(SERVER, SRV_UPDATE);
226         String clearText = Random.name();
227         console.verticalNavigation().selectPrimary(Ids.build(MESSAGING_SERVER, BRIDGE, ITEM));
228         TableFragment table = page.getBridgeTable();
229         FormFragment form = page.getBridgeCRForm();
230         EmptyState emptyState = form.emptyState();
231         table.bind(form);
232         table.select(BRIDGE_UPDATE);
233         page.getBridgeFormsTab().select(CREDENTIAL_REFERENCE_TAB);
234         emptyState.mainAction();
235         console.confirmationDialog().getPrimaryButton().click();
236 
237         AddResourceDialogFragment addResource = console.addResourceDialog();
238         addResource.getForm().text(CLEAR_TEXT, clearText);
239         addResource.add();
240 
241         console.verifySuccess();
242         new ResourceVerifier(bridgeAddress(SRV_UPDATE, BRIDGE_UPDATE), client)
243                 .verifyAttribute(CREDENTIAL_REFERENCE + "." + CLEAR_TEXT, clearText);
244 
245     }
