16  package org.jboss.hal.testsuite.test.configuration.webservice.endpoint.configuration;
17  
18  import org.apache.commons.lang3.RandomStringUtils;
19  import org.jboss.arquillian.core.api.annotation.Inject;
20  import org.jboss.arquillian.graphene.page.Page;
21  import org.jboss.hal.resources.Ids;
22  import org.jboss.hal.testsuite.Console;
23  import org.jboss.hal.testsuite.CrudOperations;
24  import org.jboss.hal.testsuite.container.WildFlyContainer;
25  import org.jboss.hal.testsuite.fixtures.WebServicesFixtures;
26  import org.jboss.hal.testsuite.page.configuration.WebServicesPage;
27  import org.jboss.hal.testsuite.test.Manatoko;
28  import org.junit.jupiter.api.BeforeAll;
29  import org.junit.jupiter.api.BeforeEach;
30  import org.junit.jupiter.api.Test;
31  import org.testcontainers.junit.jupiter.Container;
32  import org.testcontainers.junit.jupiter.Testcontainers;
33  import org.wildfly.extras.creaper.core.online.OnlineManagementClient;
34  import org.wildfly.extras.creaper.core.online.operations.Operations;
35  
36  import static org.jboss.hal.testsuite.container.WildFlyConfiguration.DEFAULT;
37  
38  @Manatoko
39  @Testcontainers
40  class PostHandlerChainTest {
41  
42      private static final String END_POINT_CONFIGURATION_EDIT = "end-point-configuration-to-be-edited-"
43              + RandomStringUtils.randomAlphanumeric(7);
44  
45      private static final WebServicesFixtures.HandlerChain POST_HANDLER_CHAIN_CREATE = new WebServicesFixtures.HandlerChain.Builder(
46              END_POINT_CONFIGURATION_EDIT)
47              .handlerChainName("post-handler-chain-to-be-created-" + RandomStringUtils.randomAlphanumeric(7))
48              .endpointConfiguration()
49              .postHandlerChain()
50              .build();
51  
52      private static final WebServicesFixtures.HandlerChain POST_HANDLER_CHAIN_EDIT = new WebServicesFixtures.HandlerChain.Builder(
53              END_POINT_CONFIGURATION_EDIT)
54              .handlerChainName("post-handler-chain-to-be-edited-" + RandomStringUtils.randomAlphanumeric(7))
55              .endpointConfiguration()
56              .postHandlerChain()
57              .build();
58  
59      private static final WebServicesFixtures.HandlerChain POST_HANDLER_CHAIN_DELETE = new WebServicesFixtures.HandlerChain.Builder(
60              END_POINT_CONFIGURATION_EDIT)
61              .handlerChainName("post-handler-chain-to-be-removed-" + RandomStringUtils.randomAlphanumeric(7))
62              .endpointConfiguration()
63              .postHandlerChain()
64              .build();
65  
66      @Container static WildFlyContainer wildFly = WildFlyContainer.standalone(DEFAULT);
67  
68      @BeforeAll
69      static void setupModel() throws Exception {
70          OnlineManagementClient client = wildFly.managementClient();
71          Operations operations = new Operations(client);
72          operations.add(WebServicesFixtures.endpointConfigurationAddress(END_POINT_CONFIGURATION_EDIT));
73          operations.add(POST_HANDLER_CHAIN_EDIT.handlerChainAddress());
74          operations.add(POST_HANDLER_CHAIN_DELETE.handlerChainAddress());
75      }
76  
77      @Inject Console console;
78      @Inject CrudOperations crudOperations;
79      @Page WebServicesPage page;
80  
81      @BeforeEach
82      void prepare() {
83          page.navigate();
84          console.verticalNavigation().selectPrimary(Ids.WEBSERVICES_ENDPOINT_CONFIG_ITEM);
85          page.getEndpointConfigurationTable().action(END_POINT_CONFIGURATION_EDIT, "Post");
86      }
87  
88      @Test
89      void create() throws Exception {
90          crudOperations.create(POST_HANDLER_CHAIN_CREATE.handlerChainAddress(),
91                  page.getEndpointConfigurationHandlerChainTable(), POST_HANDLER_CHAIN_CREATE.getHandlerChainName());
92      }
93  
94      @Test
95      void remove() throws Exception {
96          crudOperations.delete(POST_HANDLER_CHAIN_DELETE.handlerChainAddress(),
97                  page.getEndpointConfigurationHandlerChainTable(), POST_HANDLER_CHAIN_DELETE.getHandlerChainName());
98      }
99  
100     @Test
101     void editProtocolBindings() throws Exception {
102         page.getEndpointConfigurationHandlerChainTable().select(POST_HANDLER_CHAIN_EDIT.getHandlerChainName());
103         crudOperations.update(POST_HANDLER_CHAIN_EDIT.handlerChainAddress(),
104                 page.getEndpointConfigurationHandlerChainForm(), "protocol-bindings");
105     }
106 
107 }
