16  package org.jboss.hal.testsuite.fragment.finder;
17  
18  import java.util.Map;
19  
20  import org.jboss.arquillian.graphene.findby.ByJQuery;
21  import org.jboss.arquillian.graphene.fragment.Root;
22  import org.jboss.hal.testsuite.fragment.AlertFragment;
23  import org.openqa.selenium.By;
24  import org.openqa.selenium.WebElement;
25  
26  import static java.util.stream.Collectors.toMap;
27  import static org.jboss.arquillian.graphene.Graphene.createPageFragment;
28  import static org.jboss.arquillian.graphene.Graphene.waitGui;
29  import static org.jboss.hal.resources.CSS.alert;
30  import static org.jboss.hal.resources.CSS.key;
31  import static org.jboss.hal.resources.CSS.listGroup;
32  import static org.jboss.hal.resources.CSS.value;
33  
34  /** Fragment for the finder preview. Use {@link FinderFragment#preview()} to get an instance. */
35  public class FinderPreviewFragment {
36  
37      @Root protected WebElement root;
38  
39      public AlertFragment getAlert() {
40          return createPageFragment(AlertFragment.class, root.findElement(By.className(alert)));
41      }
42  
43      public Map<String, String> getMainAttributes() {
44          return getAttributeElementMap("Main Attributes").entrySet()
45                  .stream()
46                  .collect(toMap(Map.Entry::getKey, entry -> entry.getValue().getText()));
47      }
48  
49      protected Map<String, WebElement> getAttributeElementMap(String heading) {
50          By attributeSelector = ByJQuery
51                  .selector("h2:contains('" + heading + "'):visible ~ ul." + listGroup + ":eq(0) > li:visible");
52          waitGui().until().element(root, attributeSelector).is().present();
53          return root.findElements(attributeSelector).stream()
54                  .collect(toMap(attributeElement -> attributeElement.findElement(By.className(key)).getText(),
55                          attributeElement -> attributeElement.findElement(By.className(value))));
56      }
57  }
