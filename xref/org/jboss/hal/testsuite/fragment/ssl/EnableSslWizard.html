16  package org.jboss.hal.testsuite.fragment.ssl;
17  
18  import org.jboss.arquillian.drone.api.annotation.Drone;
19  import org.jboss.hal.resources.CSS;
20  import org.jboss.hal.resources.Ids;
21  import org.jboss.hal.testsuite.fragment.FormFragment;
22  import org.jboss.hal.testsuite.fragment.RadioButtonGroup;
23  import org.jboss.hal.testsuite.fragment.WizardFragment;
24  import org.openqa.selenium.By;
25  import org.openqa.selenium.WebDriver;
26  
27  import static org.jboss.arquillian.graphene.Graphene.waitGui;
28  
29  public class EnableSslWizard extends WizardFragment {
30  
31      @Drone private WebDriver browser;
32  
33      private static final String MUTUAL = "mutual",
34              KEY_STORE_STRATEGY = "key-store-strategy",
35              CONFIGURATION_FORM_ID = Ids.build(Ids.MANAGEMENT, Ids.ENABLE_SSL, "edit", Ids.FORM),
36              REVIEW_FORM_ID = Ids.build(Ids.MANAGEMENT, Ids.ENABLE_SSL, "read", Ids.FORM);
37  
38      public EnableSslWizard enableMutualAuthentication() {
39          return selectRadioButtonById(MUTUAL, "choose-mutual-yes");
40      }
41  
42      public EnableSslWizard disableMutualAuthentication() {
43          return selectRadioButtonById(MUTUAL, "choose-mutual-no");
44      }
45  
46      public EnableSslWizard createAllResources() {
47          return selectRadioButtonById(KEY_STORE_STRATEGY, "strategy-create-all");
48      }
49  
50      public EnableSslWizard obtainFromLetsEncrypt() {
51          return selectRadioButtonById(KEY_STORE_STRATEGY, "strategy-obtain-from-letsencrypt");
52      }
53  
54      public EnableSslWizard createKeyStore() {
55          return selectRadioButtonById(KEY_STORE_STRATEGY, "strategy-create-key-store");
56      }
57  
58      public EnableSslWizard reuseKeyStore() {
59          return selectRadioButtonById(KEY_STORE_STRATEGY, "strategy-reuse-key-store");
60      }
61  
62      public EnableSslWizard tryNextToConfigurationWithExpectError(String errorMessage) {
63          next();
64          By alertSelector = By.className(CSS.alertDanger);
65          waitGui().until(errorMessage).element(root, alertSelector).is().visible();
66          return this;
67      }
68  
69      public EnableSslWizard nextConfiguration() {
70          next(CONFIGURATION_FORM_ID);
71          return this;
72      }
73  
74      public EnableSslWizard nextReview() {
75          next(REVIEW_FORM_ID);
76          return this;
77      }
78  
79      public FormFragment getConfigurationForm() {
80          return getForm(CONFIGURATION_FORM_ID);
81      }
82  
83      public FormFragment getReviewForm() {
84          return getForm(REVIEW_FORM_ID);
85      }
86  
87      private EnableSslWizard selectRadioButtonById(String name, String id) {
88          new RadioButtonGroup(name, root).pickById(id);
89          return this;
90      }
91  }
