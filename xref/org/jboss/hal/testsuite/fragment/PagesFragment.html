16  package org.jboss.hal.testsuite.fragment;
17  
18  import org.jboss.arquillian.graphene.fragment.Root;
19  import org.jboss.hal.resources.CSS;
20  import org.openqa.selenium.WebElement;
21  import org.openqa.selenium.support.FindBy;
22  
23  /**
24   * Fragment for 'pages' inside views to manage deeply nested resources. Not related to real pages with a token. Typically pages
25   * in views consists of a main page with a table which contains a link to sub page(s). The sub page(s) contain a
26   * {@linkplain PageBreadcrumbFragment breadcrumb} to go back to the main page.
27   */
28  public class PagesFragment {
29  
30      @Root private WebElement root;
31      @FindBy(css = "ol." + CSS.breadcrumb + "." + CSS.page) private PageBreadcrumbFragment breadcrumb;
32  
33      public PageBreadcrumbFragment breadcrumb() {
34          return breadcrumb;
35      }
36  
37      public WebElement getRoot() {
38          return root;
39      }
40  }
