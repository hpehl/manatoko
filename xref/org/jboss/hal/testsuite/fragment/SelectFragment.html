16  package org.jboss.hal.testsuite.fragment;
17  
18  import org.jboss.arquillian.graphene.findby.ByJQuery;
19  import org.openqa.selenium.By;
20  import org.openqa.selenium.WebElement;
21  import org.openqa.selenium.support.FindBy;
22  
23  import static org.jboss.arquillian.graphene.Graphene.waitGui;
24  import static org.jboss.hal.resources.CSS.dropdownMenu;
25  import static org.jboss.hal.resources.CSS.dropdownToggle;
26  import static org.jboss.hal.resources.CSS.selectpicker;
27  import static org.jboss.hal.testsuite.Selectors.contains;
28  
29  /**
30   * Fragment for a bootstrap select element. Root should be the div w/ class {@link org.jboss.hal.resources.CSS#bootstrapSelect}
31   */
32  public class SelectFragment {
33  
34      @FindBy(css = "button." + dropdownToggle) private WebElement button;
35      @FindBy(css = "div." + dropdownMenu) private WebElement menu;
36      @FindBy(css = "select." + selectpicker) private WebElement select;
37  
38      /** Selects the specified text. */
39      public void select(String text) {
40          select(text, text);
41      }
42  
43      /** Selects the specified text which has the specified value in the internal HTML select element. */
44      public void select(String text, String value) {
45          button.click();
46          waitGui().until().element(menu).is().visible();
47  
48          menu.findElement(ByJQuery.selector("a" + contains(text))).click();
49          waitGui().until().element(select).value().equalTo(value);
50      }
51  
52      public void selectExact(String text, String value) {
53          button.click();
54          waitGui().until().element(menu).is().visible();
55          menu.findElement(By.xpath(".//span[text()=\"" + text + "\"]")).click();
56          waitGui().until().element(select).value().equalTo(value);
57      }
58  }
