16  package org.jboss.hal.testsuite.arquillian;
17  
18  import org.jboss.arquillian.core.spi.LoadableExtension;
19  import org.jboss.arquillian.drone.spi.Configurator;
20  import org.jboss.arquillian.drone.spi.Destructor;
21  import org.jboss.arquillian.drone.spi.Instantiator;
22  import org.jboss.arquillian.graphene.spi.enricher.SearchContextTestEnricher;
23  import org.jboss.hal.testsuite.environment.Environment;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  public class ManatokoExtension implements LoadableExtension {
28  
29      private static final Logger logger = LoggerFactory.getLogger(ManatokoExtension.class);
30  
31      @Override
32      public void register(final ExtensionBuilder builder) {
33          if (Environment.instance().local()) {
34              logger.info("Use Arquillian Graphene 2 extension");
35              // No need to register anything. Graphene extension found in the classpath
36              // will take over, since the Testcontainers extension isn't registered!
37          } else if (Environment.instance().remote()) {
38              logger.info("Register Testcontainers extension");
39              builder.service(Configurator.class, TestcontainersWebDriverFactory.class);
40              builder.service(Instantiator.class, TestcontainersWebDriverFactory.class);
41              builder.service(Destructor.class, TestcontainersWebDriverFactory.class);
42          }
43          logger.info("Register Manatoko enricher");
44          builder.service(SearchContextTestEnricher.class, ManatokoEnricher.class);
45      }
46  }
